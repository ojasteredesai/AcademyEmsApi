table structure

1. user_type (trainer,student,staff)
	id number,
	type_name varchar,
	created_date,
	modified_date,
	created_by int,
	modified_by int
	
2. user_detail	
	id number,
	user_code varchar,
	first_name varchar,
	last_name varchar,
	gender varchar,
	dob datetime,
	mobile_no number,
	email varchar,
	identity_id varchar,
	identity_type varchar,
	address_id fk int address,
	hashed_password varchar,
	created_date,
	modified_date,
	created_by int,
	modified_by int

3. course_type (basic,advance,trauma)
	id number,
	type_name varchar,
	description varchar,
	created_date,
	modified_date,
	created_by int,
	modified_by int	
	
4. course	
	id number,
	course_typeid fk int course_type,
	course_name varchar,
	course_description varchar,
	created_date,
	modified_date,
	created_by int,
	modified_by int		
	
5. batch	
	id number,
	course_id fk int course,
	instructor_id fk int user,
	address_id fk int user,
	duration int, - in hours
	batch_start_date date,
	batch_end_date date,
	batch_time datetime,
	capacity int,
	fees double,
	created_date,
	modified_date,
	created_by int,
	modified_by int

6. batch_details		
	id number,
	batch_id fk int batch,
	student_id fk int user,
	payment_id fk int payment,
	enrollment_date date,
	created_date,
	modified_date,
	created_by int,
	modified_by int	
	
7. batch_status (upcoming, ongoing, completed)
	id number,
	status_name varchar,
	description varchar,
	created_date,
	modified_date,
	created_by int,
	modified_by int	

8. payment_type (cash,card,upi,online)
	id number,
	type_name varchar,
	description varchar,
	created_date,
	modified_date,
	created_by int,
	modified_by int	


9. payment_status (fully paid, partially paid, not paid)
	id number,
	status_name varchar,
	description varchar,
	created_date,
	modified_date,
	created_by int,
	modified_by int	
	
10. address	
	id number,
	address_line1 varchar,
	address_line2 varchar,
	city varchar,
	state_id int,
	pincode number,
	created_date,
	modified_date,
	created_by int,
	modified_by int		

11. payment		
	id number,
	batch_enrollment_id fk int batch_enrollment,	
	payment_status_id fk int payment_status,
	payment_difference double,
	created_date,
	modified_date,
	created_by int,
	modified_by int
